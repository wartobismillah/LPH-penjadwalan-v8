var PROP_CRED_POOL   = 'CRED_POOL_V8';
var PROP_CHANNELS    = 'CHANNELS_V8';
var PROP_QUOTA_FLAGS = 'QUOTA_FLAGS_V8';
var BACKUP_SHEET_NAME = 'YT_Scheduler_v8';
var YT_API_BASE = 'https://www.googleapis.com/youtube/v3';
var OAUTH_TOKEN_URL = 'https://oauth2.googleapis.com/token';
var OAUTH_SCOPE = 'https://www.googleapis.com/auth/youtube';

// ─── STORAGE ──────────────────────────────────────────────────────────────────

function loadCredPool() {
  try { var r = PropertiesService.getScriptProperties().getProperty(PROP_CRED_POOL); return r ? JSON.parse(r) : []; } catch(e) { return []; }
}
function saveCredPool(pool) {
  PropertiesService.getScriptProperties().setProperty(PROP_CRED_POOL, JSON.stringify(pool));
  try { writeBackupSheet_(); } catch(e) {}
}
function loadChannels() {
  try { var r = PropertiesService.getScriptProperties().getProperty(PROP_CHANNELS); return r ? JSON.parse(r) : []; } catch(e) { return []; }
}
function saveChannels(chs) {
  PropertiesService.getScriptProperties().setProperty(PROP_CHANNELS, JSON.stringify(chs));
}
function loadQuotaFlags() {
  try {
    var r = PropertiesService.getScriptProperties().getProperty(PROP_QUOTA_FLAGS);
    var f = r ? JSON.parse(r) : {};
    var now = Date.now(), changed = false;
    Object.keys(f).forEach(function(k){ if(f[k] && f[k].exhaustedAt && (now-f[k].exhaustedAt)>86400000){delete f[k];changed=true;} });
    if(changed) saveQuotaFlags(f);
    return f;
  } catch(e){ return {}; }
}
function saveQuotaFlags(f) {
  PropertiesService.getScriptProperties().setProperty(PROP_QUOTA_FLAGS, JSON.stringify(f));
}

// ─── RESPONSE HELPERS ─────────────────────────────────────────────────────────

function jsonOk(d) { return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON); }
function jsonErr(m,c) { return jsonOk({success:false,error:m,errorCode:c||'ERROR'}); }

// ─── ROUTERS ──────────────────────────────────────────────────────────────────

function doGet(e) {
  try {
    var a = (e.parameter && e.parameter.action) ? e.parameter.action : 'loadData';
    if(a==='oauthCallback')     return handleOAuthCallback(e);
    if(a==='loadData')          return handleLoadData(e);
    if(a==='getAuthUrl')        return handleGetAuthUrl(e);
    if(a==='checkAuthStatus')   return handleCheckAuthStatus(e);
    if(a==='fetchVideos')       return handleFetchVideos(e);
    if(a==='getChannelInfo')    return handleGetChannelInfo(e);
    if(a==='clearQuotaFlags')   return handleClearQuotaFlags(e);
    if(a==='testQuotaRotation') return handleTestQuotaRotation(e);
    if(a==='testRevokedToken')  return handleTestRevokedToken(e);
    if(a==='testAllExhausted')  return handleTestAllExhausted(e);
    return jsonErr('Unknown action: '+a,'UNKNOWN_ACTION');
  } catch(err){ return jsonErr(err.toString(),'INTERNAL_ERROR'); }
}

function doPost(e) {
  try {
    var a = (e.parameter && e.parameter.action) ? e.parameter.action : '';
    var b = {}; try { b = JSON.parse(e.postData.contents); } catch(pe){}
    if(a==='saveCredential')   return handleSaveCredential(b);
    if(a==='removeCredential') return handleRemoveCredential(b);
    if(a==='saveChannels')     return handleSaveChannels(b);
    if(a==='scheduleVideo')    return handleScheduleVideo(b);
    if(a==='unscheduleVideo')  return handleUnscheduleVideo(b);
    if(a==='bulkSchedule')     return handleBulkSchedule(b);
    return jsonErr('Unknown POST action: '+a,'UNKNOWN_ACTION');
  } catch(err){ return jsonErr(err.toString(),'INTERNAL_ERROR'); }
}

// ─── LOAD DATA ────────────────────────────────────────────────────────────────

function handleLoadData(e) {
  var pool = loadCredPool(), channels = loadChannels(), qf = loadQuotaFlags();
  var safe = pool.map(function(c){ return {
    index:c.index, client_id:c.client_id, channel_id:c.channel_id||'',
    channel_name:c.channel_name||'', channel_handle:c.channel_handle||'',
    subscriber_count:c.subscriber_count||0, thumb_url:c.thumb_url||'',
    auth_status: qf[c.index] ? 'exhausted' : (c.auth_status||'pending'),
    added_at:c.added_at||''
  };});
  return jsonOk({success:true,credPool:safe,channels:channels});
}

// ─── GET AUTH URL ─────────────────────────────────────────────────────────────

function handleGetAuthUrl(e) {
  var idx = parseInt(e.parameter.credIndex,10), pool = loadCredPool();
  if(isNaN(idx)||idx<0||idx>=pool.length) return jsonErr('Invalid credIndex','INVALID_CRED_INDEX');
  var cred = pool[idx];
  if(!cred.client_id||!cred.client_secret) return jsonErr('Missing client_id or client_secret','INVALID_CREDENTIAL');
  var cbUrl = getScriptUrl()+'?action=oauthCallback&state='+idx;
  return jsonOk({success:true,authUrl:buildAuthUrl(cred.client_id,cbUrl,idx),credIndex:idx});
}

function buildAuthUrl(clientId,cbUrl,state) {
  return 'https://accounts.google.com/o/oauth2/v2/auth'+
    '?client_id='+encodeURIComponent(clientId)+
    '&redirect_uri='+encodeURIComponent(cbUrl)+
    '&response_type=code&scope='+encodeURIComponent(OAUTH_SCOPE)+
    '&access_type=offline&prompt=consent&include_granted_scopes=true'+
    '&state='+encodeURIComponent(String(state));
}

// ─── OAUTH CALLBACK ───────────────────────────────────────────────────────────

function handleOAuthCallback(e) {
  var code=e.parameter.code, state=e.parameter.state, error=e.parameter.error;
  if(error) return HtmlService.createHtmlOutput(cbHtml('Otorisasi Gagal','Error: '+escH(error),false));
  if(!code||state===undefined) return HtmlService.createHtmlOutput(cbHtml('Parameter Tidak Valid','Code atau state tidak ditemukan.',false));

  var idx=parseInt(state,10), pool=loadCredPool();
  if(isNaN(idx)||idx<0||idx>=pool.length) return HtmlService.createHtmlOutput(cbHtml('Index Tidak Valid','Credential index '+state+' tidak valid.',false));

  var cred=pool[idx];
  var cbUrl=getScriptUrl()+'?action=oauthCallback&state='+idx;
  try {
    var resp=UrlFetchApp.fetch(OAUTH_TOKEN_URL,{
      method:'post',contentType:'application/x-www-form-urlencoded',
      payload:{code:code,client_id:cred.client_id,client_secret:cred.client_secret,redirect_uri:cbUrl,grant_type:'authorization_code'},
      muteHttpExceptions:true
    });
    var td=JSON.parse(resp.getContentText());
    if(td.error) throw new Error((td.error_description||td.error)+'\n\nPastikan redirect URI "'+cbUrl+'" sudah terdaftar di GCP Console.');
    if(!td.refresh_token) throw new Error('Google tidak mengembalikan refresh_token.\nSolusi: Buka https://myaccount.google.com/permissions hapus akses aplikasi ini lalu otorisasi ulang.');

    pool[idx].refresh_token=td.refresh_token;
    pool[idx].access_token=td.access_token;
    pool[idx].token_expiry=Date.now()+((td.expires_in||3600)*1000);
    pool[idx].auth_status='active';

    try {
      var info=fetchChannelInfoRaw(td.access_token);
      if(info){ pool[idx].channel_id=info.id; pool[idx].channel_name=info.name; pool[idx].channel_handle=info.handle; pool[idx].subscriber_count=info.subscriberCount; pool[idx].thumb_url=info.thumbUrl; }
    } catch(ce){ Logger.log('Channel info fetch failed: '+ce); }

    saveCredPool(pool);
    return HtmlService.createHtmlOutput(cbHtml('Berhasil Terhubung!','Channel <strong>'+escH(pool[idx].channel_name||'N/A')+'</strong> ('+escH(pool[idx].channel_handle||'')+')<br>berhasil dihubungkan dengan Refresh Token permanen.<br><br>Jendela ini akan tertutup otomatis...',true));
  } catch(err) {
    pool[idx].auth_status='error';
    try { saveCredPool(pool); } catch(se){}
    return HtmlService.createHtmlOutput(cbHtml('Gagal Mendapatkan Token',escH(err.toString()),false));
  }
}

// ─── TOKEN MANAGEMENT ─────────────────────────────────────────────────────────

function getValidAccessToken(cred) {
  if(cred.access_token&&cred.token_expiry&&Date.now()<(cred.token_expiry-90000)) return cred.access_token;
  if(!cred.refresh_token) throw new Error('NO_REFRESH_TOKEN: Credential #'+cred.index+' has no refresh token. Please authorize first.');

  var resp=UrlFetchApp.fetch(OAUTH_TOKEN_URL,{
    method:'post',contentType:'application/x-www-form-urlencoded',
    payload:{refresh_token:cred.refresh_token,client_id:cred.client_id,client_secret:cred.client_secret,grant_type:'refresh_token'},
    muteHttpExceptions:true
  });
  var d=JSON.parse(resp.getContentText());
  if(d.error){
    if(d.error==='invalid_grant') throw new Error('TOKEN_REVOKED: Refresh token for credential #'+cred.index+' is invalid. User may have revoked access.');
    throw new Error('TOKEN_REFRESH_ERROR: '+(d.error_description||d.error));
  }

  var pool=loadCredPool();
  var i=pool.findIndex(function(c2){return c2.client_id===cred.client_id;});
  if(i>-1){
    pool[i].access_token=d.access_token;
    pool[i].token_expiry=Date.now()+((d.expires_in||3600)*1000);
    if(d.refresh_token) pool[i].refresh_token=d.refresh_token;
    PropertiesService.getScriptProperties().setProperty(PROP_CRED_POOL,JSON.stringify(pool));
    return pool[i].access_token;
  }
  cred.access_token=d.access_token;
  cred.token_expiry=Date.now()+((d.expires_in||3600)*1000);
  return cred.access_token;
}

// ─── CORE: ROUND-ROBIN QUOTA ROTATION ────────────────────────────────────────
// callYouTubeWithRotation(requestFn, channelId)
// - Tries each active, non-exhausted credential in order
// - On QUOTA_EXCEEDED: marks credential exhausted, rotates to next
// - On TOKEN_REVOKED: marks credential revoked, rotates to next
// - Uses LockService to prevent race conditions in concurrent calls
// - Throws ALL_QUOTA_EXHAUSTED if every credential is blocked

function callYouTubeWithRotation(requestFn, channelId) {
  var pool=loadCredPool(), qf=loadQuotaFlags();

  var candidates=pool.map(function(c,i){return{cred:c,index:i};}).filter(function(item){
    return item.cred.auth_status==='active'&&item.cred.refresh_token&&!qf[item.index];
  });

  // Prefer credential linked to this specific channel (put it first)
  if(channelId){
    var matched=candidates.filter(function(x){return x.cred.channel_id===channelId;});
    var others=candidates.filter(function(x){return x.cred.channel_id!==channelId;});
    candidates=matched.concat(others);
  }

  if(!candidates.length){
    var totalActive=pool.filter(function(c){return c.auth_status==='active';}).length;
    if(!totalActive) throw new Error('NO_ACTIVE_CREDENTIALS: Tidak ada credential yang diotorisasi. Tambahkan dan hubungkan credential di tab Settings.');
    throw new Error('ALL_QUOTA_EXHAUSTED: Semua '+totalActive+' credential telah melampaui kuota API harian. Kuota reset tengah malam Pacific Time. Tambah credential baru atau tunggu reset.');
  }

  var lastError=null;
  for(var attempt=0;attempt<candidates.length;attempt++){
    var item=candidates[attempt], cred=item.cred;
    try {
      var lock=LockService.getScriptLock(), lockOk=false;
      try { lock.waitLock(15000); lockOk=true; } catch(le){ Logger.log('Lock timeout: '+le); }
      var token;
      try { token=getValidAccessToken(cred); } finally { if(lockOk){ try{lock.releaseLock();}catch(le){} } }
      return requestFn(token,cred);
    } catch(err){
      var msg=err.toString();
      if(msg.indexOf('TOKEN_REVOKED')>-1||msg.indexOf('NO_REFRESH_TOKEN')>-1){
        var p2=loadCredPool(); if(p2[item.index]){p2[item.index].auth_status='revoked';p2[item.index].access_token='';}
        PropertiesService.getScriptProperties().setProperty(PROP_CRED_POOL,JSON.stringify(p2));
        lastError={type:'TOKEN_REVOKED',idx:item.index};
        Logger.log('[Rotate] Cred #'+item.index+' revoked → trying next');
        continue;
      }
      if(msg.indexOf('QUOTA_EXCEEDED')>-1||msg.indexOf('quotaExceeded')>-1||msg.indexOf('rateLimitExceeded')>-1||(msg.indexOf('403')>-1&&msg.indexOf('quota')>-1)){
        var qf2=loadQuotaFlags(); qf2[item.index]={exhaustedAt:Date.now()}; saveQuotaFlags(qf2);
        lastError={type:'QUOTA_EXCEEDED',idx:item.index};
        Logger.log('[Rotate] Cred #'+item.index+' quota exhausted → rotating');
        continue;
      }
      throw err; // non-recoverable
    }
  }

  if(lastError&&lastError.type==='QUOTA_EXCEEDED') throw new Error('ALL_QUOTA_EXHAUSTED: Semua credential telah melampaui kuota hari ini. Cred #'+lastError.idx+' adalah upaya terakhir.');
  if(lastError&&lastError.type==='TOKEN_REVOKED') throw new Error('ALL_TOKENS_REVOKED: Semua credential memiliki token yang dicabut. Harap otorisasi ulang di tab Settings.');
  throw new Error('ALL_CREDENTIALS_FAILED: Tidak ada credential yang berhasil melakukan panggilan API.');
}

// ─── FETCH VIDEOS ─────────────────────────────────────────────────────────────

function handleFetchVideos(e) {
  var channelId=e.parameter.channelId||'', windowDays=parseInt(e.parameter.windowDays,10)||90;
  if(!channelId) return jsonErr('channelId diperlukan','MISSING_PARAM');
  try {
    var cutoff=new Date(Date.now()-windowDays*86400000).toISOString();
    var videos=callYouTubeWithRotation(function(token){
      var allItems=[], pt='', reachedOld=false;
      do {
        var url=YT_API_BASE+'/search?part=snippet&forMine=true&type=video&maxResults=50&order=date';
        if(pt) url+='&pageToken='+encodeURIComponent(pt);
        var r1=UrlFetchApp.fetch(url,{headers:{Authorization:'Bearer '+token},muteHttpExceptions:true});
        var d1=JSON.parse(r1.getContentText());
        if(d1.error){ var e1=d1.error.message||JSON.stringify(d1.error); if(d1.error.code===403||e1.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e1); throw new Error(e1); }
        var pi=d1.items||[];
        for(var i=0;i<pi.length;i++){
          var pub=pi[i].snippet&&pi[i].snippet.publishedAt;
          if(pub&&pub<cutoff){reachedOld=true;break;}
          allItems.push(pi[i]);
        }
        pt=reachedOld?'':(d1.nextPageToken||'');
      } while(pt);
      if(!allItems.length) return [];
      var allVideos=[];
      for(var b=0;b<allItems.length;b+=50){
        var batch=allItems.slice(b,b+50);
        var ids=batch.map(function(x){return x.id&&x.id.videoId;}).filter(Boolean).join(',');
        if(!ids) continue;
        var r2=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=snippet,status,contentDetails&id='+ids,{headers:{Authorization:'Bearer '+token},muteHttpExceptions:true});
        var d2=JSON.parse(r2.getContentText());
        if(d2.error){ var e2=d2.error.message||JSON.stringify(d2.error); if(d2.error.code===403||e2.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e2); throw new Error(e2); }
        allVideos=allVideos.concat(d2.items||[]);
      }
      return allVideos;
    },channelId);
    return jsonOk({success:true,videos:videos});
  } catch(err){
    var m=err.toString(), c='FETCH_VIDEOS_ERROR';
    if(m.indexOf('ALL_QUOTA_EXHAUSTED')>-1) c='ALL_QUOTA_EXHAUSTED';
    if(m.indexOf('NO_ACTIVE_CREDENTIALS')>-1) c='NO_ACTIVE_CREDENTIALS';
    if(m.indexOf('ALL_TOKENS_REVOKED')>-1) c='ALL_TOKENS_REVOKED';
    return jsonErr(m,c);
  }
}

// ─── SCHEDULE VIDEO ───────────────────────────────────────────────────────────

function handleScheduleVideo(b) {
  if(!b.videoId) return jsonErr('videoId diperlukan','MISSING_PARAM');
  if(!b.publishAt) return jsonErr('publishAt diperlukan','MISSING_PARAM');
  try {
    callYouTubeWithRotation(function(token){
      var r1=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=status',{
        method:'put',headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'},
        payload:JSON.stringify({id:b.videoId,status:{privacyStatus:'private',publishAt:b.publishAt}}),muteHttpExceptions:true
      });
      var d1=JSON.parse(r1.getContentText());
      if(d1.error){
        var e1=d1.error.message||JSON.stringify(d1.error);
        if(d1.error.code===403||e1.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e1);
        if(d1.error.errors&&d1.error.errors[0]&&d1.error.errors[0].reason==='invalidPublishAt') throw new Error('INVALID_PUBLISH_AT: '+b.publishAt+' — Video mungkin sudah Public atau waktu < 20 menit dari sekarang.');
        throw new Error(e1);
      }
      if(b.metadata&&(b.metadata.title||b.metadata.description||(b.metadata.tags&&b.metadata.tags.length))){
        var r2=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=snippet',{
          method:'put',headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'},
          payload:JSON.stringify({id:b.videoId,snippet:{categoryId:b.metadata.categoryId||'22',title:b.metadata.title||'',description:b.metadata.description||'',tags:b.metadata.tags||[]}}),muteHttpExceptions:true
        });
        var d2=JSON.parse(r2.getContentText());
        if(d2.error){ var e2=d2.error.message||JSON.stringify(d2.error); if(d2.error.code===403||e2.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e2); throw new Error(e2); }
      }
    },b.channelId||'');
    return jsonOk({success:true,videoId:b.videoId,publishAt:b.publishAt});
  } catch(err){
    var m=err.toString(), c='SCHEDULE_ERROR';
    if(m.indexOf('ALL_QUOTA_EXHAUSTED')>-1) c='ALL_QUOTA_EXHAUSTED';
    if(m.indexOf('INVALID_PUBLISH_AT')>-1) c='INVALID_PUBLISH_AT';
    if(m.indexOf('TOKEN_REVOKED')>-1) c='TOKEN_REVOKED';
    if(m.indexOf('NO_ACTIVE_CREDENTIALS')>-1) c='NO_ACTIVE_CREDENTIALS';
    return jsonErr(m,c);
  }
}

// ─── UNSCHEDULE VIDEO ─────────────────────────────────────────────────────────

function handleUnscheduleVideo(b) {

  try {
    callYouTubeWithRotation(function(token){
      var r=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=status',{
        method:'put',headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'},
        payload:JSON.stringify({id:b.videoId,status:{privacyStatus:'private'}}),muteHttpExceptions:true
      });
      var d=JSON.parse(r.getContentText());
      if(d.error){ var em=d.error.message||JSON.stringify(d.error); if(d.error.code===403||em.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+em); throw new Error(em); }
    },b.channelId||'');
    return jsonOk({success:true,videoId:b.videoId});
  } catch(err){ return jsonErr(err.toString(),'UNSCHEDULE_ERROR'); }
}

// ─── BULK SCHEDULE ────────────────────────────────────────────────────────────

function handleBulkSchedule(b) {
  var items=b.items, channelId=b.channelId||'';
  if(!Array.isArray(items)||!items.length) return jsonErr('items array diperlukan','MISSING_PARAM');
  var ok=[], fail=[], allExhausted=false;
  for(var i=0;i<items.length;i++){
    var item=items[i];
    if(!item.videoId||!item.publishAt){ fail.push({videoId:item.videoId||'?',error:'Missing videoId or publishAt',code:'MISSING_PARAM'}); continue; }
    if(allExhausted){ fail.push({videoId:item.videoId,error:'Skipped: ALL_QUOTA_EXHAUSTED',code:'ALL_QUOTA_EXHAUSTED'}); continue; }
    try {
      callYouTubeWithRotation(function(token){
        var r1=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=status',{
          method:'put',headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'},
          payload:JSON.stringify({id:item.videoId,status:{privacyStatus:'private',publishAt:item.publishAt}}),muteHttpExceptions:true
        });
        var d1=JSON.parse(r1.getContentText());
        if(d1.error){ var e1=d1.error.message||JSON.stringify(d1.error); if(d1.error.code===403||e1.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e1); if(d1.error.errors&&d1.error.errors[0]&&d1.error.errors[0].reason==='invalidPublishAt') throw new Error('INVALID_PUBLISH_AT: '+e1); throw new Error(e1); }
        if(item.metadata&&(item.metadata.title||item.metadata.description||(item.metadata.tags&&item.metadata.tags.length))){
          var r2=UrlFetchApp.fetch(YT_API_BASE+'/videos?part=snippet',{
            method:'put',headers:{Authorization:'Bearer '+token,'Content-Type':'application/json'},
            payload:JSON.stringify({id:item.videoId,snippet:{categoryId:item.metadata.categoryId||'22',title:item.metadata.title||'',description:item.metadata.description||'',tags:item.metadata.tags||[]}}),muteHttpExceptions:true
          });
          var d2=JSON.parse(r2.getContentText());
          if(d2.error){ var e2=d2.error.message||JSON.stringify(d2.error); if(d2.error.code===403||e2.toLowerCase().indexOf('quota')>-1) throw new Error('QUOTA_EXCEEDED: '+e2); throw new Error(e2); }
        }
      },channelId);
      ok.push(item.videoId);
    } catch(err){
      var m=err.toString(), c='SCHEDULE_ERROR';
      if(m.indexOf('ALL_QUOTA_EXHAUSTED')>-1){c='ALL_QUOTA_EXHAUSTED';allExhausted=true;}
      if(m.indexOf('INVALID_PUBLISH_AT')>-1) c='INVALID_PUBLISH_AT';
      if(m.indexOf('TOKEN_REVOKED')>-1) c='TOKEN_REVOKED';
      fail.push({videoId:item.videoId,error:m,code:c});
    }
  }
  return jsonOk({success:fail.length===0,successIds:ok,failures:fail,successCount:ok.length,failCount:fail.length,allExhausted:allExhausted});
}

// ─── SAVE CREDENTIAL ──────────────────────────────────────────────────────────

function handleSaveCredential(b) {
  var cid=b.client_id, cs=b.client_secret;
  if(!cid||!cs) return jsonErr('client_id dan client_secret diperlukan.','MISSING_CREDENTIAL');
  if(cid.indexOf('.apps.googleusercontent.com')===-1) return jsonErr('Format client_id tidak valid. Harus berakhir dengan .apps.googleusercontent.com','INVALID_FORMAT');
  if(cs.length<10) return jsonErr('client_secret terlalu pendek atau tidak valid.','INVALID_SECRET');
  var pool=loadCredPool();
  for(var i=0;i<pool.length;i++){ if(pool[i].client_id===cid) return jsonErr('Client ID ini sudah ada di pool (index #'+i+'). Gunakan Reconnect jika ingin otorisasi ulang.','DUPLICATE_CREDENTIAL'); }
  var nc={index:pool.length,client_id:cid,client_secret:cs,refresh_token:'',access_token:'',token_expiry:0,channel_id:'',channel_name:'',channel_handle:'',subscriber_count:0,thumb_url:'',auth_status:'pending',added_at:new Date().toISOString()};
  pool.push(nc);
  saveCredPool(pool);
  var credIndex=pool.length-1;
  var cbUrl=getScriptUrl()+'?action=oauthCallback&state='+credIndex;
  return jsonOk({success:true,credIndex:credIndex,authUrl:buildAuthUrl(cid,cbUrl,credIndex)});
}

// ─── REMOVE CREDENTIAL ────────────────────────────────────────────────────────

function handleRemoveCredential(b) {
  var idx=parseInt(b.credIndex,10), pool=loadCredPool();
  if(isNaN(idx)||idx<0||idx>=pool.length) return jsonErr('Invalid credIndex','INVALID_CRED_INDEX');
  pool.splice(idx,1);
  pool.forEach(function(c,i){c.index=i;});
  saveCredPool(pool);
  var f=loadQuotaFlags(), nf={};
  Object.keys(f).forEach(function(k){ var ki=parseInt(k,10); if(ki<idx) nf[ki]=f[k]; else if(ki>idx) nf[ki-1]=f[k]; });
  saveQuotaFlags(nf);
  return jsonOk({success:true});
}

// ─── GET CHANNEL INFO ─────────────────────────────────────────────────────────

function handleGetChannelInfo(e) {
  var idx=parseInt(e.parameter.credIndex,10), pool=loadCredPool();
  if(isNaN(idx)||idx<0||idx>=pool.length) return jsonErr('Invalid credIndex','INVALID_CRED_INDEX');
  var cred=pool[idx];
  if(cred.auth_status!=='active'||!cred.refresh_token) return jsonErr('Credential belum diotorisasi','NOT_AUTHORIZED');
  try {
    var token=getValidAccessToken(cred), info=fetchChannelInfoRaw(token);
    if(info){ pool[idx].channel_id=info.id; pool[idx].channel_name=info.name; pool[idx].channel_handle=info.handle; pool[idx].subscriber_count=info.subscriberCount; pool[idx].thumb_url=info.thumbUrl; saveCredPool(pool); }
    return jsonOk({success:true,channel:info});
  } catch(err){ return jsonErr(err.toString(),'GET_CHANNEL_INFO_ERROR'); }
}

function fetchChannelInfoRaw(accessToken) {
  var r=UrlFetchApp.fetch(YT_API_BASE+'/channels?part=snippet,statistics&mine=true',{headers:{Authorization:'Bearer '+accessToken},muteHttpExceptions:true});
  var d=JSON.parse(r.getContentText());
  if(d.error||!d.items||!d.items.length) return null;
  var ch=d.items[0], snap=ch.snippet||{}, stats=ch.statistics||{}, thumbs=snap.thumbnails||{};
  var thumb=thumbs.medium||thumbs.default||thumbs.high||thumbs.maxres||{};
  var handle=snap.customUrl||''; if(handle&&handle[0]!=='@') handle='@'+handle;
  return {id:ch.id,name:snap.title||'',handle:handle,subscriberCount:parseInt(stats.subscriberCount,10)||0,thumbUrl:thumb.url||''};
}

  var pool=loadCredPool();

  if(pool.length<2) return jsonErr('Need at least 2 credentials to demonstrate rotation.','INSUFFICIENT_CREDENTIALS');

  var reqCount=0, log=[];

    try {
      callYouTubeWithRotation(function(token,cred){

          log.push({req:reqCount,credIdx:cred.index,ch:cred.channel_name||'(no channel)',status:'⛔ QUOTA_EXCEEDED [SIMULATED] → rotating...'});

    }

  // Clean up simulated quota flags
  var f=loadQuotaFlags(); delete f[credIdx]; saveQuotaFlags(f);

  return jsonOk({success:true,testType:'quota_rotation',targetCred:credIdx,failAfter:failAfter,log:log,
